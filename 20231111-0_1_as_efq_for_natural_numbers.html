0 = 1 as EfQ for Natural Numbers

1. Satz

Es gilt

\begin{align*} 0 \equiv S(0) \rightarrow \forall_{x,y} x \equiv y \end{align*}

2. Beweis

Sei

\begin{align*} P (x) \coloneqq 0 \equiv x \end{align*}

Dann gilt \(P(0) = 0 \equiv 0\) wegen

\begin{prooftree} \infer1[ $\mathrm{Refl}$ ]{$0 \equiv 0$} \infer1[ $\mathrm{def.}$]{P(0)} \hypo{[u: P x]} \infer1[ $\mathrm{def}$]{$0 \equiv x$} \infer1[ $\mathrm{Suc}$]{S(0) \equiv S(x)} \hypo{$0 \equiv S(0)$} \infer2[ $\mathrm{Trans}$]{0 \equiv S(x)} \infer1[ $\mathrm{def}$]{P(S(x))} \infer2[ $\mathrm{Ind} (*)$]{$\forall_x P(x)$} \infer1[ $\rightarrow^+$]{$0 \equiv S(0) \rightarrow \forall_x P(x)$} \end{prooftree}

Damit folgt auch

\begin{prooftree} \hypo{v:[ $\forall_x P(x)$]} \hypo{t} \infer1[ $\forall^-$]{P(t)} \infer1[ $\mathrm{def}$]{0 \equiv t} \infer1[ $\mathrm{Sym}$]{t \equiv 0} \hypo{v: [ $\forall_y P(y)$]} \hypo{u} \infer1[ $\forall^-$]{P(u)} \infer1[ $\mathrm{def}$]{0 \equiv u} \infer2[ $\mathrm{Trans}$]{t \equiv u} \infer1[ $\forall^+ \mathrm{var cond.}$]{$\forall_x x \equiv u$} \infer1[ $\forall^+ \mathrm{var cond.}$]{$\forall_{x,y} x \equiv y$} \infer1[ $\rightarrow^+$ v]{ $\forall_x P(x) \rightarrow \forall_{x,y} x \equiv y$} \end{prooftree}

zusammengesetzt ergibt sich

\begin{prooftree} \hypo{w: $0 \equiv S(0)$} \ellipsis{N}{$\forall_x P(x)$} \hypo{u: $\forall_x P(x)$} \ellipsis{M}{$\forall_{x,y} x \equiv y$} \infer1[ $\rightarrow^+$ u]{$ \forall_x P(x) \rightarrow \forall_{x,y} x \equiv y$ \infer2[ $\rightarrow^-$]{ $\forall_{x,y} x \equiv y$} \infer1[ $\rightarrow^+$ w]{ $0 \equiv S(0) \rightarrow \forall_{x,y} x \equiv y$} \end{prooftree}
