1.2. Berechnung

\begin{align*} \begin{pmatrix} s_1 \cdot \widehat{i} \\ s_2 \cdot \widehat{j} \end{pmatrix} \rightarrow \begin{pmatrix} s_1 \cdot x_1 + s_2 \cdot x_2 \\ s_1 \cdot y_1 + s_2 \cdot y_2 \end{pmatrix} \\ s_1 \widehat{i} + s_2 \widehat{j} \rightarrow s_1 \widehat{i_2} + s_2 \widehat{j_2} \end{align*} \begin{align*} \begin{pmatrix} x_1 & y_1 \\ x_2 & y_2 \end{pmatrix} \end{align*} \begin{align*} \begin{pmatrix} a & b \\ c & d \end{pmatrix} \begin{pmatrix} x \\ y \end{pmatrix} = x \begin{pmatrix} a \\ c \end{pmatrix} + y \begin{pmatrix} b \\ d \end{pmatrix} \end{align*}
