2. Einzelne bedingte Wahrscheinlichkeiten

\begin{equation} P_B(A) = P_{\bar{B}(A) \end{equation}

2.1. Beweis

2.1.1. Lemma

\begin{align*} B = P(A \cap B) \dot\cup (\bar{A} \cap B) \\ \end{align*}

2.1.2. Umformung

2.1.2.1. a)
\begin{align*} B = P(A \cap B) \dot\cup (\bar{A} \cap B) \\ P(B) = P(A \cap B) + P(\bar{A} \cap B) \\ P(\bar{A} \cap B) = P(B) - P(A \cap B) \end{align*}
2.1.2.2. b)
\begin{equation} P_A(b) = P_{\bar(A}}(B) \Leftrightarrow \\ \frac{P(A \cap B)}{P(A)} = \frac{P(\bar{A} \cap B)}{P(\bar{A})} \Leftrightarrow \\ P(A \cap B) \cdot P(\bar{A}) = P(A) \cdot P(\bar{A} \cap B) \Leftrightarrow \\ P(A \cap B) \cdot (1 - P(\bar{A})) = P(A) \cdot (P(B) - P(A \cap B)) \Leftrightarrow \\ P(A \cap B) - P(A) \cdot P(A \cap b) = P(A) \cdot P(B) - P(A) \cdot P(A \cap B) \Leftrightarrow \\ P(A \cap B) = P(A) \cdot P(B) \end{equation}
